\block{Why use UD for L2 corpora?}{
    \vskip 5pt
    \begin{itemize}
        \item existing parsers allow for \textbf{faster, semi-automatic annotation}
        \item rich morphosyntactic UD annotation supports the \textbf{study of L2 \\grammatical patterns}
        \item UD provides a uniform annotation layer that enables \textbf{cross-lingual \\comparisons}:
        \begin{itemize}
            \item between a learner's L1 and L2
            \item between different L2s
            \item between standard and learner language
        \end{itemize}
    \end{itemize}
    \vskip 15pt  
}

\block{L1-L2 Treebanks}{
    \vskip 20pt
    \textbf{Parallel treebanks} where \textbf{learner sentences} are \textbf{paired with \\ \textit{correction hypotheses}}.
    
    \bigskip\bigskipamount
    L1-L2 treebanks can be an even better basis for:
    \begin{itemize}
        \item \textbf{grammatical error retrieval and analysis}
        \item automatic \textbf{feedback comment generation}
    \end{itemize}
    \vskip 15pt  
}

\block{Overall Project Goals}{
    %This project focuses on the latter case. More specifically, its first objective is to use L1-L2 treebanks, a type of parallel UD corpus where learner productions are contrasted with corrections, for grammatical error retrieval and analysis. Moreover, the intention is to experiment with dependency parsing of L2 material and use the resulting models to investigate whether error patterns extracted from automatic analyses can be used as a basis for feedback comment generation.
    \begin{itemize}
        \item improving \textbf{UD guidelines for L2} (Swedish) \textbf{treebanks}
        \item \color{sborange}creating an \textbf{L1-L2 Swedish treebank}\color{black}
        \item training \textbf{parsing models for L2 material}
        \item developing \textbf{tools for parallel (L1-L2) treebanks}
    \end{itemize}
}

\block{SweLL-UD: a Treebank of L2 Swedish}{
    \innerblock{Source Corpus}{
        \textbf{SweLL-gold}, aka the Swedish Learner Language corpus*:
        \begin{itemize}
            \item \textbf{genre}: essays (miscellaneous topics)
            \item \textbf{learners}: adult L2 Swedish students with various language \\backgrounds and proficiency levels
            \item \textbf{annotation}: manual correction, error tagging, pseudonymization \\and normalization (minimal edits)
            \item \textbf{size}: 502 essays ($\to$ 5000+ parallel sentences)
            \item \textbf{license}: CLARIN-ID -PRIV \underline{-NORED} -BY \\(but the data can be redistributed as long as some metadata is \\removed and full essays cannot be reconstructed)
        \end{itemize}
    }
    \small{* part of the Manually Annotated Corpora CLARIN Resource Family\newline}
    \vskip 5pt
    \large
    \innerblock{Project Status and Plan}{
    \vskip 10pt
        \begin{enumerate}
            \item \textbf{preprocessing}:
            \begin{itemize}
                \item sentence pair extraction ($\checkmark$)
                \item automatic pre-annotation ($\checkmark$)
            \end{itemize}
        \item \textbf{manual validation} of a 500-sentence test set $\parallel$ \textbf{guidelines \\development} (ongoing)
        \item \textbf{test set} release (planned in 2025) - 2 versions: 
        \begin{itemize}
            \item sentence-shuffled version at \color{gublue}\url{universaldependencies.org}\color{black}
            \item full-essay version with all metadata released as the source corpus 
        \end{itemize}
        \item gradual annotation and release of a \textbf{development and training set}
        \end{enumerate}
    }
}
